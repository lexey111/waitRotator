Bonus part: entire UI lock


    

        If you need a full-screen lock: click here to show wait layer for 5 second.
    


    

        You can call function showWait(duration, ms) to show wait layer for required period of time.
    


    

        If you call showWait() without duration parameter the layer will be shown for infinite time. You have to
        call waitHide() function explicitly, e.g. after end of AJAX call.
    


    

        Layout (add this just before </body> and don't forget to attach wait-rotator.css):
    

    
<div id="wait-rotator-fullscreen" style="display: none">
    <div class="wait-rotator-wrapper">
        <div class="wait-rotator-circle wrc-slow wrc-thick wrc-1-slice"></div>
        <span>Please wait...</span>
    </div>
</div>
    

    

        Two simple JS functions to show the wait layer and to hide it.
    


    

        JavaScript support code (vanilla, you can use jQuery instead):
    

    
/**
* Show fullscreen wait layer
* @param {Number} duration in ms - optional
*/
var wait_layer = document.getElementById("wait-rotator-fullscreen");
var wait_timer;
function showWait(duration) {
    // show the wait layer
    wait_layer.style.display = "block";
    wait_layer.style.opacity = 1;

    // start new countdown if any
    clearTimeout(wait_timer);

    // check is wait duration set at all...
    if ((typeof duration != "undefined") && (!isNaN(parseInt(duration)) && isFinite(duration))) {
        // yes - well, let's start countdown timer
        wait_timer = setTimeout(function () {
            hideWait();
        }, duration);
    }
}

/**
* Hide wait layer (fadeOut)
*/
function hideWait() {
    wait_layer.style.opacity = 1;
    (function fade() {
        if ((wait_layer.style.opacity -= .2) <= .2) {
            wait_layer.style.display = "none";
        } else {
            setTimeout(fade, 40);
        }
    })();
}
    

    

        And use example —
    


    

        Code to call full-screen wait layer onClick (vanilla):
    

    
// assign call for full-screen wait layer (5 second)
// when user click on the demo link
document.getElementById("fullscreen-wait").onclick = function (e) {
    e.preventDefault();
    showWait(5000);
    return false;
};
